1. Write a recursive function to print first N natural numbers

#include<conio.h>

{

    scanf("%d",&num);
    printN(num);

}
void printN(int n)
{
    if(n>0)
    {
        printN(n-1);

    else
    {

7. Write a recursive function to print squares of first N natural numbers

#include<stdio.h>
#include<conio.h>
int square(int);
int main()
{
    int num,sq;
    printf("enter numbers to print:");
    scanf("%d",&num);
    sq=square(num);
    printf("square of %d is:%d",num,sq);
    getch();
}
int square(int n)
{
    static int  s=1;
    static int c=0;
    if(c==2)
        return s;
    s*=n;
    c++;
    square(n);
}

8. Write a recursive function to print binary of a given decimal number

#include<stdio.h>
#include<conio.h>
int binary(int);
int main()
{
    int num,bin;
    printf("enter numbers to print:");
    scanf("%d",&num);
    bin=binary(num);
    printf("binary number is: %d",bin);
    getch();
}
int binary(int n)
{
    static int b=0;
    static int p=1;
    if(n==0)
        return b;
    b+=((n%2)*p);
    p*=10;
    binary(n/2);
}

10. Write a recursive function to print reverse of a given number

#include<stdio.h>
#include<conio.h>
int reverse(int);
int main()
{
    int num,r;
    printf("enter numbers to print:");
    scanf("%d",&num);
    r=reverse(num);
    printf("reverse number is: %d",r);
    getch();
}
int reverse(int n)
{
    static int rev=0;
    if(n==0)
        return rev;
    rev=rev*10+(n%10);
    reverse(n/10);
}
